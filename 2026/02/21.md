```
return (

    <Link to="/path" or  "/path/* ">リンク名</Link>     /// path/* はpathの下に階層があれば*をつける

        <Route path="/" element={<コンポーネント />}>
        <Route path="/" element={<コンポーネント />}>
        <Route path="/" element={<コンポーネント />}>
    </Routes>
</BrowserRouter>
);
```
