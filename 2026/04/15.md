- 複数の情報を追加する（createMany()）
```
export async function main() {
  const createMany = await prisma.user.createMany({
    data: [
      {
        name: "Bob",
        email: "bob@prisma.io",
        city: "",
        country: "",
        pets: "",
      },
      {
        name: "Yewande",
        email: "yewande@prisma.io",
        city: "",
        country: "",
        pets: "",
      },
    ],
    skipDuplicates: true, // Skip records with duplicate unique fields
  });
  // Returns: { count: 2 }
  console.log(createMany);
}

main();
```
