type Profile = {
    name: string;
    age?: number;
    zipcode: number;
};

type PartialType = Partial<Profile>;             // 全ての項目に?がつく
type RequiredType = Required<Profile>;    // 全ての項目が必須

- Recordユーティリティ型
```
type Prefectures = 'Tokyo' | 'Chiba' ;     // ’Shiga’を追加すればエラー解消

type Covid19InfectionInfo = {
    kanji_name: string;
    confirmed_cases: number;
};

const covid19Japan: Record<Prefectures, Covid19InfectionInfo> = {.   // Recordでそれぞれ用意するはずの型を省略できる
    Tokyo: { kanji_name: '東京', confirmed_cases: 1960 },
    Chiba: { kanji_name: '千葉', confirmed_cases: 800 },
    Shiga: { kanji_name: '滋賀', confirmed_cases: 50 },    // エラーを検知
}
```
