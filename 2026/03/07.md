for文
```
	i := 0
	for i <=3 {
		fmt.Println(i)
		i++
	}

	for i:=0; i<4; i++ {
		fmt.Println(i)	
	}

	arr := [3]int{1, 2, 3}
	for i, v := range arr {
		fmt.Println(i,v)        // 0 1    1 2   2 3　が出力
	}
```
