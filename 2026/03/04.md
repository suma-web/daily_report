- オブジェクトの型を作ることができる`interface`
```
type ObjectType = {
    name: string;
    age: number;
}

interface ObjectInterface {
    name: string;
    age: number; 
}

let object: ObjectInterface = {
    name: "Hana",
    age: 13 
}
```
- 型を共有
```
let batter = {
    daritu: number 
}
let pitcher = {
    syubiritu: number
}
let twoWay = batter & pitcher;  // batter と pitcher の型を持った新たな型ができる

```
- `unknown`型は型が予測しづらい時に仮称的につけ型を絞ってある程度把握できた型を設定していく。
- `union`型（異なる型を使用できる）
```
let value:number | string = 1;   //number型だけのところをstringも可能に
value = 'foo';
console.log(value);
```
- リテラル型（より厳密に型を指定できる。`union`型の応用）
```
let dayOfTheWeek: '日'|'月'| '火'|'水'|'木'|'金'|'土' = '日';
dayOfTheWeek = '月';
dayOfTheWeek = '14'    // エラー
```
- `enum`型（順番に数値が割り振られる）
```
enum Months {
    January,
    February,
    March,
    April,
    May
}

Months.January;     // 0 が出力
```
- 付け足したい時
```
enum Months {
    January = "0",
    February = "1",
    March = "2",
    April = "3",
    May = "4"
}

enum Months{      // enum型で追加する要素を記述
    June = "5"
}

console.log(Months.June);     // 5 が出力
```

func Div(x, y int) (int, int) {         // 引数はまとめて型を指定できる。返り値が複数ある場合も指定できる
	q := x / y
